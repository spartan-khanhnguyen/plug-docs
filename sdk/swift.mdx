## Overview

The plugnpay-swift SDK provides a secure, server-side interface for developers to integrate their Swift applications with the PlugnPay service. The library is built from the ground up using Swift's modern concurrency features (async/await) to offer a high-performance, non-blocking API. It leverages Swift's strong type system and Codable protocol for robust, safe, and easy-to-use data handling.

**Requirements:** Swift 5.7+ (Swift Concurrency)
**Distribution:** Swift Package Manager (SPM)

## Configuration & Initialization

The SDK client is an actor to ensure thread-safe access to its properties and methods. It is configured via an initializer that accepts a configuration object.

### API Key Authentication

Authentication is handled via an API key that can be configured in two ways:

**Environment Variable (Recommended):**
```bash
export PLUGNPAY_API_KEY="pnp_sk_xxxxxx"
```

**Direct Configuration:**
```swift
import PlugnPay

do {
    // Preferred: Initialize by reading PLUGNPAY_API_KEY from the environment
    // This initializer can throw an error if the key is not found
    let client = try PlugnPayClient.fromEnvironment()
    
    // Alternatively, initialize with explicit config
    let config = PlugnPayClient.Config(
        apiKey: "pnp_sk_xxxxxx",
    )
    let clientWithOptions = try PlugnPayClient(config: config)
} catch {
    print("Failed to initialize PlugnPay client: \(error)")
}
```

### Configuration Options

```swift
    public let apiKey: String // API key for authentication

        self.apiKey = apiKey
    }
```

## Core Methods

All methods that interact with the REST API are async and throws, designed to be called with `try await`. Models are defined as struct types conforming to Codable for seamless JSON serialization.

```swift

    print("Failed to get customer: \(error)")
}
```

## Real-time Event Handling

The SDK provides an AsyncThrowingStream interface for handling real-time events via Server-Sent Events (SSE). This is the standard, modern Swift pattern for handling a sequence of values and potential errors over time, integrating perfectly with async/await.

### AsyncStream Interface

The client provides a method that returns an AsyncThrowingStream.

```swift
let eventStream = client.listenForEvents()
```

**Method:** `func listenForEvents() -> AsyncThrowingStream<Event, Error>`

**Behavior:**
- The stream is "cold" — the SSE connection is only established when the stream is first iterated over (e.g., in a `for try await` loop)
- The connection is automatically closed when the Task iterating over the stream is cancelled
- Errors in the stream are thrown as Swift errors that can be caught with do-catch blocks

### Event Structure

The SDK uses an enum with associated values for event data payloads, providing compile-time type safety:

```swift
// Generic event wrapper
public struct Event: Decodable {
    public let eventId: String
    public let eventType: String
    public let createdAt: String
    public let data: EventData
}

// Enum for type-safe event data
public enum EventData: Decodable {
    case subscription(SubscriptionEvent)
    case invoice(InvoiceEvent)
    case unknown // Fallback for future event types
}

// Payload models
public struct SubscriptionEvent: Decodable {
    public let customer: Customer
    public let subscription: Subscription
}

public struct InvoiceEvent: Decodable {
    public let customer: Customer
    public let invoice: Invoice
}
```

### Advanced Event Handling with Structured Concurrency

```swift
import PlugnPay
import Foundation

actor PlugnPayEventHandler {
    private let client: PlugnPayClient
    private var eventTask: Task<Void, Error>?
    
    init(client: PlugnPayClient) {
        self.client = client
    }
    
    func start() {
        // Cancel any existing event handling task
        eventTask?.cancel()
        
        eventTask = Task {
            do {
                let eventStream = client.listenForEvents()
                
                for try await event in eventStream {
                    // Process events concurrently but safely within the actor
                    Task {
                        await self.processEvent(event)
                    }
                }
            } catch {
                print("Event stream error: \(error)")
                // Implement exponential backoff retry logic
                try await Task.sleep(nanoseconds: 5_000_000_000) // 5 seconds
                await start() // Restart listening
            }
        }
    }
    
    func stop() {
        eventTask?.cancel()
        eventTask = nil
    }
    
    private func processEvent(_ event: Event) async {
        do {
            print("Processing event \(event.eventId): \(event.eventType)")
            
            switch event.data {
            case .subscription(let data):
                await handleSubscriptionEvent(event.eventType, data: data)
            case .invoice(let data):
                await handleInvoiceEvent(event.eventType, data: data)
            case .unknown:
                print("Unknown event type: \(event.eventType)")
            }
        } catch {
            print("Error processing event \(event.eventId): \(error)")
        }
    }
    
    private func handleSubscriptionEvent(_ eventType: String, data: SubscriptionEvent) async {
        switch eventType {
        case "subscription.created":
            print("New \(data.subscription.productId) subscription for \(data.customer.email)")
            await provisionAccount(for: data.customer.id)
            await sendWelcomeEmail(to: data.customer.email)
        case "subscription.updated":
            print("Subscription updated for \(data.customer.email): \(data.subscription.status)")
            await updateUserPermissions(for: data.customer.id, status: data.subscription.status)
        case "subscription.deleted":
            print("Subscription cancelled for \(data.customer.email)")
            await revokeAccess(for: data.customer.id)
        default:
            print("Unhandled subscription event: \(eventType)")
        }
    }
    
    private func handleInvoiceEvent(_ eventType: String, data: InvoiceEvent) async {
        switch eventType {
        case "invoice.payment.succeeded":
            let amount = Double(data.invoice.amountPaid) / 100.0
            print("Payment of $\(amount) succeeded for \(data.customer.email)")
            await sendPaymentReceipt(to: data.customer.email, invoiceUrl: data.invoice.url)
        case "invoice.payment.failed":
            print("Payment failed for \(data.customer.email)")
            await sendPaymentFailedNotification(to: data.customer.id, invoiceUrl: data.invoice.url)
            await pauseServiceAccess(for: data.customer.id)
        default:
            print("Unhandled invoice event: \(eventType)")
        }
    }
    
    // Helper methods - implement according to your needs
    private func provisionAccount(for customerId: String) async {
        print("Provisioning account for customer \(customerId)")
        // Implement account provisioning logic
    }
    
    private func sendWelcomeEmail(to email: String) async {
        print("Sending welcome email to \(email)")
        // Implement welcome email sending
    }
    
    private func updateUserPermissions(for customerId: String, status: String) async {
        print("Updating permissions for customer \(customerId) to \(status)")
        // Implement permission updates
    }
    
    private func revokeAccess(for customerId: String) async {
        print("Revoking access for customer \(customerId)")
        // Implement access revocation
    }
    
    private func sendPaymentReceipt(to email: String, invoiceUrl: String?) async {
        print("Sending payment receipt to \(email)")
        // Implement receipt sending
    }
    
    private func sendPaymentFailedNotification(to customerId: String, invoiceUrl: String?) async {
        print("Sending payment failed notification to customer \(customerId)")
        // Implement payment failure notification
    }
    
    private func pauseServiceAccess(for customerId: String) async {
        print("Pausing service access for customer \(customerId)")
        // Implement service access pausing
    }
}

// Usage
@main

    static func main() async {
        do {
            let client = try PlugnPayClient.fromEnvironment()
            let eventHandler = PlugnPayEventHandler(client: client)
            
            await eventHandler.start()
```

## Error Handling

The SDK uses Swift's native error handling with typed exceptions:

```swift
enum PlugnPayError: Error {
    case invalidAPIKey
    case networkError(Error)
    case decodingError(Error)
    case httpError(statusCode: Int, message: String)
}

// Usage with proper error handling
do {
    let customer = try await client.getCustomer(email: "test@example.com")
    print("Customer: \(customer)")
} catch let error as PlugnPayError {
    switch error {
    case .invalidAPIKey:
        print("Invalid API key provided")
    case .networkError(let underlyingError):
        print("Network error: \(underlyingError)")
    case .decodingError(let underlyingError):
        print("Failed to decode response: \(underlyingError)")
    case .httpError(let statusCode, let message):
        print("HTTP error \(statusCode): \(message)")
    }
} catch {
    print("Unexpected error: \(error)")
}
```

## Complete Integration Example

```swift
import PlugnPay
import Foundation

class PlugnPayService {
    private let client: PlugnPayClient
    
    init() throws {
        self.client = try PlugnPayClient.fromEnvironment()
    }
    
    func createCustomerAndCheckout(email: String, productId: String) async -> CheckoutResult {
        do {
            // Get or create customer
            let customer = try await client.getCustomer(email: email)
            print("Customer ID: \(customer.id)")
            
            // Create checkout session
            let session = try await client.createCheckoutSession(
                with: CreateCheckoutSessionRequest(
                    customerId: customer.id,
                    productId: productId,
                    successUrl: "https://yourapp.com/success",
                    cancelUrl: "https://yourapp.com/cancel"
                )
            )
            
            return .success(
                customerId: customer.id,
                checkoutUrl: session.url
            )
        } catch {
            return .error("Checkout creation failed: \(error)")
        }
    }
    
    func checkUserAccess(customerId: String, featureId: String) async -> AccessResult {
        do {
            let access = try await client.checkAccess(
                customerId: customerId,
                featureId: featureId
            )
            
            if access.hasAccess {
                // Track usage for metered features
                _ = try await client.trackUsage(
                    with: TrackUsageRequest(
                        customerId: customerId,
                        featureId: featureId,
                        value: 1
                    )
                )
                
                // Get updated usage stats
                let usage = try await client.getUsage(
                    customerId: customerId,
                    featureId: featureId
                )
                
                return .granted(
                    usage: usage.totalUsage,
                    remaining: usage.remaining
                )
            } else {
                return .denied
            }
        } catch {
            return .error("Access check failed: \(error)")
        }
    }
}

// Result types for better error handling
enum CheckoutResult {
    case success(customerId: String, checkoutUrl: String)
    case error(String)
}

enum AccessResult {
    case granted(usage: Int, remaining: Int?)
    case denied
    case error(String)
}

// Usage example
@main
struct IntegrationApp {
    static func main() async {
        do {
            let plugnPayService = try PlugnPayService()
            
            // Create checkout for a customer
            let checkoutResult = await plugnPayService.createCustomerAndCheckout(
                email: "customer@example.com",
                productId: "premium-plan"
            )
            
            switch checkoutResult {
            case .success(let customerId, let checkoutUrl):
                print("Redirect to: \(checkoutUrl)")
                
                // Later, check if user has access to features
                let accessResult = await plugnPayService.checkUserAccess(
                    customerId: customerId,
                    featureId: "premium-features"
                )
                
                switch accessResult {
                case .granted(let usage, let remaining):
                    print("Feature access granted. Usage: \(usage)")
                    if let remaining = remaining {
                        print("Remaining: \(remaining)")
                    }
                case .denied:
                    print("Upgrade required for this feature")
                case .error(let message):
                    print("Error: \(message)")
                }
                
            case .error(let message):
                print("Checkout failed: \(message)")
            }
        } catch {
            print("Failed to initialize PlugnPay service: \(error)")
        }
    }
}
```

## Vapor Integration

For Vapor applications, you can integrate PlugnPay in your routes:

```swift
import PlugnPay
import Vapor

extension Application {
    var plugnPay: PlugnPayClient {
        get {
            guard let client = storage[PlugnPayKey.self] else {
                fatalError("PlugnPay client not configured")
            }
            return client
        }
        set {
            storage[PlugnPayKey.self] = newValue
        }
    }
}

struct PlugnPayKey: StorageKey {
    typealias Value = PlugnPayClient
}

// Configure PlugnPay
func configure(_ app: Application) throws {
    app.plugnPay = try PlugnPayClient.fromEnvironment()
}

// Route handlers
func routes(_ app: Application) throws {
    app.post("checkout") { req async throws -> CheckoutResponse in
        let request = try req.content.decode(CheckoutRequest.self)
        
        let customer = try await app.plugnPay.getCustomer(email: request.email)
        let session = try await app.plugnPay.createCheckoutSession(
            with: CreateCheckoutSessionRequest(
                customerId: customer.id,
                productId: request.productId,
                successUrl: request.successUrl,
                cancelUrl: request.cancelUrl
            )
        )
        
        return CheckoutResponse(checkoutUrl: session.url)
    }
}

struct CheckoutRequest: Content {
    let email: String
    let productId: String
    let successUrl: String
    let cancelUrl: String?
}

struct CheckoutResponse: Content {
    let checkoutUrl: String
}
```

## iOS/macOS Specific Considerations

### Background Tasks

For iOS applications handling events:

```swift
import BackgroundTasks

class AppDelegate: UIResponder, UIApplicationDelegate {
    var eventHandler: PlugnPayEventHandler?
    
    func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
        
        BGTaskScheduler.shared.register(forTaskWithIdentifier: "com.yourapp.plugnpay-events", using: nil) { task in
            self.handlePlugnPayEvents(task: task as! BGAppRefreshTask)
        }
        
        return true
    }
    
    func handlePlugnPayEvents(task: BGAppRefreshTask) {
        let client = try! PlugnPayClient.fromEnvironment()
        let handler = PlugnPayEventHandler(client: client)
        self.eventHandler = handler
        
        task.expirationHandler = {
            Task {
                await handler.stop()
            }
            task.setTaskCompleted(success: false)
        }
        
        Task {
            await handler.start()
            task.setTaskCompleted(success: true)
        }
    }
}
```
